SELECT AB.DIA, T.ID AS TANQUE, B.ID AS BOMBA, SUM(AB.VALOR_ABASTECIDO) AS VALOR_DIARIO
FROM BOMBA B
INNER JOIN ABASTECIMENTO AB
ON B.ID = AB.ID_BOMBA
INNER JOIN TANQUE T
ON T.ID = AB.ID_TANQUE
WHERE AB.DIA BETWEEN '05.04.2017' AND '20.04.2017'
GROUP BY AB.DIA, T.ID, B.ID
ORDER BY AB.DIA

SELECT SUM(AB.VALOR_ABASTECIDO) AS VALOR_PERIODO
FROM BOMBA B, TANQUE T, ABASTECIMENTO AB
WHERE B.ID = AB.ID_BOMBA AND T.ID = AB.ID_TANQUE AND AB.DIA BETWEEN '05.04.2017' AND '25.04.2017';
